class PageHeaderMistakesAndWarnings extends StatefulWidget {
  final int pageNumber;

  PageHeaderMistakesAndWarnings({Key? key, required this.pageNumber})
      : super(key: key);

  @override
  State<PageHeaderMistakesAndWarnings> createState() =>
      _PageHeaderMistakesAndWarningsState();
}

class _PageHeaderMistakesAndWarningsState
    extends State<PageHeaderMistakesAndWarnings> {
  int mistakes = 0;
  int warnings = 0;
  final wordsColorsMap = WordColorMap();
  @override
  void initState() {
    // TODO: implement initState
    super.initState();
    fetchMistakes();
    // quranProvider.mistakes
    //     .firstWhereOrNull((element) => element.pageNumber == widget.pageNumber);
  }

  void fetchMistakes() {
    bool iswerd = Provider.of<QuranProvider>(context, listen: false).isWerd;

    var ff;
    if (iswerd) {
      ff = Provider.of<QuranProvider>(context, listen: false)
          .werdMistakes
          .lastWhereOrNull(
              (element) => element.pageNumber == widget.pageNumber);
    } else {
      ff = Provider.of<QuranProvider>(context, listen: false)
          .mistakes
          .firstWhereOrNull(
              (element) => element.pageNumber == widget.pageNumber);
    }
    if (mounted) {
      setState(() {
        mistakes = ff?.mistakes == 0 || ff?.mistakes == null ? 0 : ff?.mistakes;
        warnings = ff?.warnings == 0 || ff?.warnings == null ? 0 : ff?.warnings;
      });
    }
  }

  var textColor = 0xffae8f74;
  @override
  Widget build(BuildContext context) {
    log("rendered header");
    return Row(
      children: [
        warnings > 0
            ? Row(
                children: [
                  Text(
                    warnings.toString(),
                    style: TextStyle(
                        fontSize: 12,
                        color: Color(textColor),
                        fontFamily: "montserrat"),
                  ),
                  const SizedBox(width: 1),
                  Container(
                    decoration: BoxDecoration(
                        shape: BoxShape.circle,
                        color: Color(int.parse(MistakesColors.warning))),
                    height: 8,
                    width: 8,
                  ),
                ],
              )
            : const SizedBox(width: 13),
        const SizedBox(width: 4),
        mistakes > 0
            ? Row(
                children: [
                  Text(
                    mistakes.toString(),
                    style: TextStyle(
                        fontSize: 12,
                        color: Color(textColor),
                        fontFamily: "montserrat"),
                  ),
                  const SizedBox(width: 1),
                  Container(
                    decoration: BoxDecoration(
                        shape: BoxShape.circle,
                        color: Color(int.parse(MistakesColors.mistake))),
                    height: 8,
                    width: 8,
                  ),
                ],
              )
            : const SizedBox(width: 13),
      ],
    );
    // return Text("a");
  }
}
